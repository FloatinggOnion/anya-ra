/**
 * Parse LLM response into draft sections.
 * Splits by markdown headers (##) or double newlines.
 */
export function parseDraftSections(content: string): DraftSection[] {
  // Split by markdown headers or paragraphs
  const sections = content
    .split(/\n##\s+|\n\n/)
    .filter(s => s.trim().length > 50) // Ignore short fragments
    .map((content, i) => ({
      id: crypto.randomUUID(),
      content: content.trim(),
      accepted: false,
      rejected: false,
      editedContent: null,
      position: i
    }))
  
  return sections
}

import { parseDraftSections, type DraftSection } from '$lib/types/chat'

// Update initial state
const initialState: ChatState = {
  messages: [],
  isStreaming: false,
  error: null,
  draftSections: [],
  currentChatId: null
}

export const chatState = writable<ChatState>(initialState)

export async function requestDraft(
  prompt: string,
  provider: LLMProvider,
  options: ChatOptions = {}
) {
  const draftPrompt = `Generate a draft section for a literature review based on this request: "${prompt}"\n\nProvide 2-3 paragraphs with clear section headers. Format with markdown (## Header).`
  
  await streamChat(draftPrompt, provider, options)
  
  // After streaming completes, parse draft sections
  chatState.update(s => {
    const lastMessage = s.messages[s.messages.length - 1]
    if (lastMessage && lastMessage.role === 'assistant') {
      const sections = parseDraftSections(lastMessage.content)
      lastMessage.draftSections = sections
      return {
        ...s,
        draftSections: sections,
        messages: [...s.messages]
      }
    }
    return s
  })
}

export function acceptDraft(sectionId: string) {
  chatState.update(s => {
    const section = s.draftSections.find(d => d.id === sectionId)
    if (section) {
      section.accepted = true
      section.rejected = false
    }
    return { ...s, draftSections: [...s.draftSections] }
  })
}

export function rejectDraft(sectionId: string) {
  chatState.update(s => {
    const section = s.draftSections.find(d => d.id === sectionId)
    if (section) {
      section.rejected = true
      section.accepted = false
    }
    return { ...s, draftSections: [...s.draftSections] }
  })
}

export function editDraft(sectionId: string, editedContent: string) {
  chatState.update(s => {
    const section = s.draftSections.find(d => d.id === sectionId)
    if (section) {
      section.editedContent = editedContent
      section.accepted = true // Auto-accept edited drafts
    }
    return { ...s, draftSections: [...s.draftSections] }
  })
}

export function getAcceptedDrafts(): DraftSection[] {
  const state = get(chatState)
  return state.draftSections
    .filter(d => d.accepted && !d.rejected)
    .sort((a, b) => a.position - b.position)
}

export async function exportAcceptedDrafts(): Promise<string> {
  const drafts = getAcceptedDrafts()
  return drafts
    .map(d => d.editedContent || d.content)
    .join('\n\n')
}
```

import { invoke } from '@tauri-apps/api/core'
import { get } from 'svelte/store'
import { workspace } from '$lib/stores/workspace'
import { chatState } from '$lib/stores/chat'
import type { ChatHistory } from '$lib/types/chat'

export async function saveChatHistory(chatId: string): Promise<void> {
  const ws = get(workspace)
  if (!ws) throw new Error('No workspace selected')
  
  const state = get(chatState)
  const history: ChatHistory = {
    id: chatId,
    workspacePath: ws.path,
    messages: state.messages,
    draftSections: state.draftSections,
    createdAt: state.currentChatId === chatId ? new Date().toISOString() : state.messages[0]?.timestamp || new Date().toISOString(),
    updatedAt: new Date().toISOString()
  }
  
  await invoke('save_chat_file', {
    workspacePath: ws.path,
    chatId,
    content: JSON.stringify(history, null, 2)
  })
}

export async function loadChatHistory(chatId: string): Promise<ChatHistory | null> {
  const ws = get(workspace)
  if (!ws) throw new Error('No workspace selected')
  
  try {
    const content = await invoke<string>('load_chat_file', {
      workspacePath: ws.path,
      chatId
    })
    return JSON.parse(content)
  } catch (err) {
    console.warn('Failed to load chat history:', err)
    return null
  }
}

export async function createNewChat(): Promise<string> {
  const chatId = crypto.randomUUID()
  await saveChatHistory(chatId)
  return chatId
}

export async function listChatFiles(): Promise<string[]> {
  const ws = get(workspace)
  if (!ws) return []
  
  try {
    return await invoke<string[]>('list_chat_files', {
      workspacePath: ws.path
    })
  } catch (err) {
    console.warn('Failed to list chat files:', err)
    return []
  }
}

// Auto-save debounce
let saveTimeout: number | undefined
export function autoSaveChatHistory(chatId: string) {
  if (saveTimeout) clearTimeout(saveTimeout)
  saveTimeout = setTimeout(() => {
    saveChatHistory(chatId).catch(console.error)
  }, 2000) as any
}

use std::fs;
use std::path::PathBuf;

#[tauri::command]
pub fn save_chat_file(
    workspace_path: String,
    chat_id: String,
    content: String,
) -> Result<(), String> {
    let mut chat_dir = PathBuf::from(&workspace_path);
    chat_dir.push("chats");
    
    // Create chats directory if it doesn't exist
    fs::create_dir_all(&chat_dir)
        .map_err(|e| format!("Failed to create chats directory: {}", e))?;
    
    chat_dir.push(format!("{}.json", chat_id));
    
    fs::write(&chat_dir, content)
        .map_err(|e| format!("Failed to write chat file: {}", e))?;
    
    Ok(())
}

#[tauri::command]
pub fn load_chat_file(
    workspace_path: String,
    chat_id: String,
) -> Result<String, String> {
    let mut chat_path = PathBuf::from(&workspace_path);
    chat_path.push("chats");
    chat_path.push(format!("{}.json", chat_id));
    
    fs::read_to_string(&chat_path)
        .map_err(|e| format!("Failed to read chat file: {}", e))
}

#[tauri::command]
pub fn list_chat_files(workspace_path: String) -> Result<Vec<String>, String> {
    let mut chat_dir = PathBuf::from(&workspace_path);
    chat_dir.push("chats");
    
    if !chat_dir.exists() {
        return Ok(Vec::new());
    }
    
    let entries = fs::read_dir(&chat_dir)
        .map_err(|e| format!("Failed to read chats directory: {}", e))?;
    
    let mut chat_ids = Vec::new();
    for entry in entries {
        if let Ok(entry) = entry {
            if let Some(name) = entry.file_name().to_str() {
                if name.ends_with(".json") {
                    let id = name.trim_end_matches(".json").to_string();
                    chat_ids.push(id);
                }
            }
        }
    }
    
    Ok(chat_ids)
}
```
