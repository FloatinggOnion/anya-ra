Create `src/lib/stores/llm-provider.ts`:
```typescript
import { writable, get } from 'svelte/store'
import type { LLMProvider } from '$lib/types/llm'
import { createProvider } from '$lib/services/llm/providers'
import { invoke } from '@tauri-apps/api/core'

export interface ProviderState {
  type: 'ollama' | 'openai'
  provider: LLMProvider | null
  apiKey: string | null
  error: string | null
}

const initialState: ProviderState = {
  type: 'ollama',
  provider: null,
  apiKey: null,
  error: null
}

export const providerState = writable<ProviderState>(initialState)

export async function initializeProvider() {
  try {
    // Try loading OpenAI key from keystore
    const apiKey = await invoke<string | null>('load_api_key', { service: 'openai' })
    
    // Default to Ollama
    const provider = createProvider('ollama')
    const available = await provider.isAvailable()
    
    if (available) {
      providerState.set({ type: 'ollama', provider, apiKey, error: null })
    } else {
      providerState.set({ 
        type: 'ollama', 
        provider: null, 
        apiKey, 
        error: 'Ollama not running. Please start Ollama and refresh.' 
      })
    }
  } catch (err) {
    providerState.update(s => ({ 
      ...s, 
      error: err instanceof Error ? err.message : 'Unknown error' 
    }))
  }
}

export async function switchProvider(type: 'ollama' | 'openai', apiKey?: string) {
  try {
    const provider = type === 'ollama' 
      ? createProvider('ollama')
      : createProvider('openai', { apiKey })
    
    const available = await provider.isAvailable()
    if (!available) {
      throw new Error(`${type} provider not available`)
    }
    
    providerState.set({ type, provider, apiKey: apiKey || null, error: null })
  } catch (err) {
    providerState.update(s => ({ 
      ...s, 
      error: err instanceof Error ? err.message : 'Unknown error' 
    }))
  }
}

export async function setApiKey(service: 'openai', apiKey: string) {
  try {
    await invoke('save_api_key', { service, apiKey })
    providerState.update(s => ({ ...s, apiKey, error: null }))
  } catch (err) {
    providerState.update(s => ({ 
      ...s, 
      error: err instanceof Error ? err.message : 'Failed to save API key' 
    }))
  }
}
```

Create `src-tauri/src/commands/keystore.rs`:
```rust
use tauri::State;
use tauri_plugin_store::StoreExt;

#[tauri::command]
pub fn save_api_key(
    app: tauri::AppHandle,
    service: String,
    api_key: String,
) -> Result<(), String> {
    let store = app.store("keystore.json")
        .map_err(|e| format!("Failed to access store: {}", e))?;
    
    let key = format!("api_key_{}", service);
    store.set(&key, api_key);
    store.save().map_err(|e| format!("Failed to save: {}", e))?;
    
    Ok(())
}

#[tauri::command]
pub fn load_api_key(
    app: tauri::AppHandle,
    service: String,
) -> Result<Option<String>, String> {
    let store = app.store("keystore.json")
        .map_err(|e| format!("Failed to access store: {}", e))?;
    
    let key = format!("api_key_{}", service);
    Ok(store.get(&key).and_then(|v| v.as_str().map(String::from)))
}
```

<script lang="ts">
  import { providerState, switchProvider, setApiKey } from '$lib/stores/llm-provider'
  
  let showSettings = $state(false)
  let apiKeyInput = $state('')
  let selectedProvider = $state<'ollama' | 'openai'>('ollama')
  
  $effect(() => {
    selectedProvider = $providerState.type
    apiKeyInput = $providerState.apiKey || ''
  })
  
  async function handleProviderChange() {
    if (selectedProvider === 'openai' && !apiKeyInput.trim()) {
      return // Don't switch without API key
    }
    
    await switchProvider(selectedProvider, apiKeyInput || undefined)
  }
  
  async function handleSaveKey() {
    if (!apiKeyInput.trim()) return
    await setApiKey('openai', apiKeyInput)
  }
</script>

<div class="provider-settings">
  <button class="toggle" onclick={() => showSettings = !showSettings}>
    <span class="provider-label">
      {$providerState.type === 'ollama' ? '🦙 Ollama' : '🤖 OpenAI'}
    </span>
    <span class="toggle-icon">{showSettings ? '▼' : '▶'}</span>
  </button>
  
  {#if showSettings}
    <div class="settings-panel">
      <div class="radio-group">
        <label>
          <input
            type="radio"
            value="ollama"
            bind:group={selectedProvider}
            onchange={handleProviderChange}
          />
          Ollama (Local)
        </label>
        <label>
          <input
            type="radio"
            value="openai"
            bind:group={selectedProvider}
            onchange={handleProviderChange}
          />
          OpenAI (Cloud)
        </label>
      </div>
      
      {#if selectedProvider === 'openai'}
        <div class="api-key-input">
          <input
            type="password"
            bind:value={apiKeyInput}
            placeholder="sk-..."
            onblur={handleSaveKey}
          />

        </div>
      {/if}
      
      {#if $providerState.error}
        <div class="error">{$providerState.error}</div>
      {/if}
    </div>
  {/if}
</div>
